Algorithme multiUnN
variables n, mult, i: entiers

Début
        afficher ("Calcul de la multiplication de 1 à un nombre n\n");
	afficher("Entrez un nombre: ");
	lire(n);
	mult<-1;
	pour (i<-; i<=n; i++)
		    afficher(i);
		    si(i<n) alors
		        afficher("+");
		    finsi
		    mult<-mult*i;
	finpour
	afficher(" = ", mult);
Fin
